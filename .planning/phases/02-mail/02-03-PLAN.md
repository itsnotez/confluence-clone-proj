List<MailMessageDto.Response> findByAccount(String spaceKey, Long accountId, Long userId, String userRole, List<Long> groupIds):
  1. space = spaceRepo.findBySpaceKeyAndDeletedAtIsNull(spaceKey).orElseThrow(SPACE_NOT_FOUND)
  2. 권한 체크: permissionService.resolveSpacePermission(space.getId(), userId, userRole, groupIds)
     → "NONE"이면 throw PERMISSION_DENIED
  3. account = mailAccountRepo.findByIdAndSpaceId(accountId, space.getId()).orElseThrow(MAIL_ACCOUNT_NOT_FOUND)
  4. return mailMessageRepo.findByMailAccountIdOrderByReceivedAtDesc(accountId).stream().map(Response::from).toList()

MailMessageDto.ConvertResponse convertToPage(String spaceKey, Long accountId, Long msgId, Long userId, String userRole, List<Long> groupIds):
  1. space = spaceRepo... SPACE_NOT_FOUND
  2. isSpaceAdmin 체크: PermissionService.isSpaceAdmin 또는 userRole.contains("SITE_ADMIN")
     → 아니면 PERMISSION_DENIED
  3. account = mailAccountRepo.findByIdAndSpaceId(accountId, space.getId()).orElseThrow(MAIL_ACCOUNT_NOT_FOUND)
  4. msg = mailMessageRepo.findByIdAndMailAccountId(msgId, accountId).orElseThrow(MAIL_MESSAGE_NOT_FOUND)
  5. if (msg.getLinkedContentId() != null) throw ALREADY_CONVERTED
  6. 제목: msg.getSubject(), body JSON: {"type":"doc","content":[{"type":"paragraph","content":[{"type":"text","text":"<bodyText>"}]}]}
     단, bodyText가 null이면 빈 doc
  7. Content content = Content.builder()
       .spaceId(space.getId()).parentId(null).title(msg.getSubject())
       .type("PAGE").status("PUBLISHED").authorId(userId).createdAt(now()).updatedAt(now())
       .build()
     content = contentRepo.save(content)
  8. ContentVersion version = ContentVersion.builder()
       .contentId(content.getId()).versionNumber(1).body(bodyJson)
       .authorId(userId).createdAt(now())
       .build()
     version = contentVersionRepo.save(version)
  9. content.setCurrentVersionId(version.getId())
     content.setUpdatedAt(now())
     contentRepo.save(content)
  10. msg.setLinkedContentId(content.getId())
      msg.setStatus("CONVERTED")
      mailMessageRepo.save(msg)
  11. return new ConvertResponse(content.getId(), content.getTitle(), "메일이 페이지로 변환되었습니다")
