       export const adminApi = {
         getStats: () => api.get('/admin/stats'),
         getAuditLogs: (params) => api.get('/admin/audit-logs', { params })
       }

    2. frontend/src/router/index.js의 /admin 라우트에 meta: { requiresAdmin: true } 추가:
       { path: '/admin', component: () => import('@/views/admin/AdminDashboardView.vue'), meta: { requiresAdmin: true } }

    3. router.beforeEach 가드에 requiresAdmin 체크 추가:
       if (to.meta.requiresAdmin) {
         const auth = useAuthStore()
         if (!auth.user || auth.user.role !== 'SITE_ADMIN') return '/spaces'
       }
       단, auth.user가 null일 수 있으므로 auth.isLoggedIn 체크 후 role 체크 순서를 지킨다.
       auth.user가 localStorage accessToken은 있지만 user 객체가 없는 경우(새로고침 후)를 위해 await auth.fetchMe()를 고려하지 않는다 — 가드에서 비동기 처리 복잡도 증가. user가 null이면 /login으로 이미 리다이렉트되므로 /spaces로 보내는 것으로 충분하다.
  </action>
