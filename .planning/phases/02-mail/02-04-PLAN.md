mail.js store (Composition API, defineStore('mail', () => {...})):

state:
  const accounts = ref([])
  const messages = ref([])
  const selectedMessage = ref(null)
  const loading = ref(false)
  const error = ref(null)

actions:
  async fetchAccounts(spaceKey):
    loading.value = true
    const res = await mailAccountApi.getAccounts(spaceKey)
    accounts.value = res.data.data
    loading.value = false

  async fetchMessages(spaceKey, accountId):
    loading.value = true
    const res = await mailMessageApi.getMessages(spaceKey, accountId)
    messages.value = res.data.data
    loading.value = false

  async convertToPage(spaceKey, accountId, msgId):
    const res = await mailMessageApi.convertToPage(spaceKey, accountId, msgId)
    // 성공 후 해당 메시지 status 업데이트
    const idx = messages.value.findIndex(m => m.id === msgId)
    if (idx >= 0) messages.value[idx].linkedContentId = res.data.data.contentId
    return res.data.data

  function selectMessage(msg): selectedMessage.value = msg

return { accounts, messages, selectedMessage, loading, error, fetchAccounts, fetchMessages, convertToPage, selectMessage }

MailBoxView.vue — 레이아웃: 좌측 계정 셀렉트 + DxDataGrid(메시지), 우측 미리보기 패널

<script setup>:
import { ref, computed, onMounted, watch } from 'vue'
import { useRoute } from 'vue-router'
import { useMailStore } from '@/stores/mail'
import { DxDataGrid, DxColumn, DxPaging, DxSelection } from 'devextreme-vue/data-grid'
import { DxButton } from 'devextreme-vue/button'
import { DxSelectBox } from 'devextreme-vue/select-box'
import notify from 'devextreme/ui/notify'

const route = useRoute()
const mailStore = useMailStore()
const spaceKey = computed(() => route.params.spaceKey)
const selectedAccountId = ref(null)
const showPreview = ref(false)

onMounted(async () => {
  await mailStore.fetchAccounts(spaceKey.value)
  if (mailStore.accounts.length > 0) {
    selectedAccountId.value = mailStore.accounts[0].id
  }
})

watch(selectedAccountId, async (id) => {
  if (id) await mailStore.fetchMessages(spaceKey.value, id)
})

function onRowClick({ data }) {
  mailStore.selectMessage(data)
  showPreview.value = true
}

async function handleConvert(msg) {
  try {
    const result = await mailStore.convertToPage(spaceKey.value, selectedAccountId.value, msg.id)
    notify({ message: result.message, type: 'success', displayTime: 3000 })
  } catch (e) {
    const msg409 = e.response?.status === 409 ? '이미 변환된 메일입니다' : '변환 실패'
    notify({ message: msg409, type: 'error', displayTime: 3000 })
  }
}
